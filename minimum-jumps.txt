int minJumps(int arr[], int n){
  
      int des = 0, count = 0, pos = 0;
       for(int i = 0; i < n -1; i++)
       {
           des = max(des, i+arr[i]);
           if(i == pos)
           {
               count++;
               pos = des;
           }
               
           if(i>=des) return -1;
       }
       return count;
    }
